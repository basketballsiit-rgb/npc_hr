function doPost(e) {
  try {
    const json = JSON.parse(e.postData.contents);
    const event = json.events[0];
    
    if (event && event.type === 'message' && event.message.type === 'text') {
      const replyToken = event.replyToken;
      const userMessage = event.message.text.trim();
      const userId = event.source.userId;
      
      const lowerMessage = userMessage.toLowerCase().replace(/\s+/g, '');
      if (lowerMessage === 'ขอid' || lowerMessage === 'id') {
        replyLineMessage(replyToken, "รหัส LINE User ID ของคุณคือ:\n\n" + userId);
      }
    }
  } catch (error) {
    console.error("Error in doPost webhook: " + error.message);
  }
  return ContentService.createTextOutput("OK");
}

function replyLineMessage(replyToken, text) {
  const url = "https://api.line.me/v2/bot/message/reply";
  const payload = {
    "replyToken": replyToken,
    "messages": [
      {
        "type": "text",
        "text": text
      }
    ]
  };
  
  const options = {
    "method": "post",
    "headers": {
      "Authorization": "Bearer " + CONFIG.LINE_ACCESS_TOKEN,
      "Content-Type": "application/json"
    },
    "payload": JSON.stringify(payload)
  };
  
  UrlFetchApp.fetch(url, options);
}

function registerUser(userData) {
  try {

    const usersData = usersSheet.getDataRange().getValues();
    
    const userExists = usersData.some(row => row[3] === userData.username);

    const userId = Utilities.getUuid();
    let role = 'user';
    let status = 'pending';
    if (usersData.length <= 1) {
      role = 'admin';
      status = 'approved';
    }

    usersSheet.appendRow([
        userId, 
        userData.fullName, 
        userData.position, 
        userData.username, 
        userData.password, 
        role, 
        status, 
        userData.lineUserId || '' 
    ]);

    return { success: true, message: 'ลงทะเบียนสำเร็จ! กรุณารอผู้ดูแลระบบอนุมัติการใช้งาน' };

    return { success: false, message: 'เกิดข้อผิดพลาด: ' + error.message };
  }
}

    } catch (error) {
        return { success: false, message: 'Error: ' + error.message };
    }
}

// --- 5. การคำนวณวันลา ---
function calculateOfficialDays(startStr, endStr, leaveType, isHalfDay) {
  if (isHalfDay) return 0.5;
  if (!startStr || !endStr) return 0;

  const startDate = new Date(startStr);
  const endDate = new Date(endStr);
  startDate.setHours(0,0,0,0); endDate.setHours(0,0,0,0);
  if (endDate < startDate) return 0;

  const workDayTypes = ['ลาป่วย', 'ลากิจส่วนตัว', 'ลาพักผ่อน', 'ลาไปช่วยเหลือภริยาที่คลอดบุตร'];
  let count = 0;
  if (workDayTypes.includes(leaveType)) {
    const holidays = getHolidaysList(); 
    let loopDate = new Date(startDate);
    while (loopDate <= endDate) {
      const dayOfWeek = loopDate.getDay();
      const dateString = Utilities.formatDate(loopDate, "GMT+7", "yyyy-MM-dd");
      if (dayOfWeek !== 0 && dayOfWeek !== 6 && !holidays.includes(dateString)) {
        count++;
      }
      loopDate.setDate(loopDate.getDate() + 1);
    }
  } else {
    const diffTime = Math.abs(endDate - startDate);
    count = Math.ceil(diffTime / (1000 * 60 * 60 * 24)) + 1;
  }
  return count;
}

function getHolidaysList() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheet = ss.getSheetByName(CONFIG.SHEET.HOLIDAYS);
  if (!sheet || sheet.getLastRow() <= 1) return [];
  const data = sheet.getRange(2, 1, sheet.getLastRow() - 1, 1).getValues(); 
  return data.map(row => formatDateToYyyyMmDd(row[0])).filter(d => d !== '');
}

  const sheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
  const lastRow = sheet.getLastRow();
  if (lastRow <= 1) return null;
  const data = sheet.getRange(2, 1, lastRow - 1, 19).getValues(); 
  for (let i = data.length - 1; i >= 0; i--) {
    if (data[i][1] === userId && data[i][18] === 'อนุมัติ') {
      return {
        leaveType: data[i][6], startDate: formatDateToYyyyMmDd(data[i][8]), 
        endDate: formatDateToYyyyMmDd(data[i][9]), totalDays: data[i][10]               
      };
    }
  }
  return null;
}

function sendLinePushMessage(to, messageText) {
  if (!to || to.trim() === '') return;
  if (!CONFIG.LINE_ACCESS_TOKEN || CONFIG.LINE_ACCESS_TOKEN.includes('ใส่_Channel')) return;

  const url = "https://api.line.me/v2/bot/message/push";
  const payload = { "to": to, "messages": [{ "type": "text", "text": messageText }] };
  const options = {
    "method": "post",
    "headers": { "Authorization": "Bearer " + CONFIG.LINE_ACCESS_TOKEN, "Content-Type": "application/json" },
    "payload": JSON.stringify(payload),
    "muteHttpExceptions": true
  };
  try { UrlFetchApp.fetch(url, options); } catch (e) { console.error("Push Error: " + e.toString()); }
}

function processLeaveRequest(approvalData) {
    try {
        const leaveSheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
        const data = leaveSheet.getDataRange().getValues();
        const rowIndex = data.findIndex(row => row[0] === approvalData.leaveId);
        if (rowIndex === -1) return { success: false, message: 'ไม่พบข้อมูลใบลา' };
        
        const leaveRowData = data[rowIndex];
        const adminSignatureUrl = uploadBase64ImageToDrive(approvalData.adminSignature, `adminsig_${approvalData.leaveId}`);
        const approvalDate = new Date().toLocaleDateString('th-TH', { year: 'numeric', month: 'long', day: 'numeric' });
        
        leaveSheet.getRange(rowIndex + 1, 19, 1, 6).setValues([[
            approvalData.status, approvalData.comment || '', adminSignatureUrl, 
            approvalData.adminUser.fullName, approvalData.adminUser.position, approvalDate
        ]]);

        const pdfUrl = generatePdf(leaveRowData, approvalData, adminSignatureUrl, approvalDate);
        leaveSheet.getRange(rowIndex + 1, 25).setValue(pdfUrl);

        const requestorUserId = leaveRowData[1];
        const requestorLineId = getUserLineIdBySystemId(requestorUserId);
        
        if (requestorLineId) {
            let emoji = approvalData.status === 'อนุมัติ' ? '✅' : '❌';
            let msg = `${emoji} ผลการพิจารณา\nเรื่อง: ${leaveRowData[6]}\nผล: ${approvalData.status}\nโดย: ${approvalData.adminUser.fullName}`;
            if(pdfUrl && approvalData.status === 'อนุมัติ') msg += `\n\n📄 ดาวน์โหลด: ${pdfUrl}`;
            sendLinePushMessage(requestorLineId, msg);
        }

        return { success: true, message: `ดำเนินการ '${approvalData.status}' เรียบร้อย`, pdfUrl: pdfUrl };
    } catch (e) {
        return { success: false, message: 'Error: ' + e.message };
    }
}

function cancelLeaveRequest(leaveId, userId) {
    try {

        const data = sheet.getDataRange().getValues();

            if (ownerId !== requestUserId) return { success: false, message: 'คุณไม่มีสิทธิ์ยกเลิกใบลาของคนอื่น' };

    } catch (e) {
        return { success: false, message: 'เกิดข้อผิดพลาด: ' + e.toString() };
    }
}

function deleteLeaveHistory(leaveId) {
  try {
    const sheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
    const data = sheet.getDataRange().getValues();
    
    // ค้นหาแถวจากล่างขึ้นบน
    for (let i = data.length - 1; i >= 1; i--) {
      if (String(data[i][0]) === String(leaveId)) {
        // ลบไฟล์ใน Drive (Optional)
        try {
           if(data[i][17]) DriveApp.getFileById(data[i][17]).setTrashed(true); // Signature
           if(data[i][20]) DriveApp.getFileById(data[i][20]).setTrashed(true); // Admin Sig
           if(data[i][24]) {
              const pid = data[i][24].match(/[-\w]{25,}/);
              if(pid) DriveApp.getFileById(pid[0]).setTrashed(true); // PDF
           }
        } catch(e) {}

        // ลบแถว
        sheet.deleteRow(i + 1);
        return { success: true, message: 'ลบข้อมูลสำเร็จ' };
      }
    }
    return { success: false, message: 'ไม่พบรายการที่ต้องการลบ' };
  } catch (e) {
    return { success: false, message: 'Error: ' + e.toString() };
  }
}

  try {
    const usersSheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.USERS);
    const leaveSheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
    const totalStaff = usersSheet.getLastRow() - 1;
    const leaveData = leaveSheet.getLastRow() > 1 ? leaveSheet.getDataRange().getValues().slice(1) : [];
    let approved=0, pending=0, rejected=0;
    const leaveTypeCounts = {};
    const monthlyLeaveCounts = Array(12).fill(0);
    
    leaveData.forEach(row => {
      const status = row[18];
      if (status === 'อนุมัติ') approved++;
      else if (status === 'รอการอนุมัติ') pending++;
      else if (status === 'ไม่อนุมัติ') rejected++;

      if (row[6]) leaveTypeCounts[row[6]] = (leaveTypeCounts[row[6]] || 0) + 1;
      
      const dateStr = formatDateToYyyyMmDd(row[8]);
      if (dateStr) monthlyLeaveCounts[parseInt(dateStr.substring(5, 7), 10) - 1]++;
    });

    const recentLeaves = leaveData.slice(-10).reverse().map(row => ({
      fullName: row[2], leaveType: row[6], startDate: formatDateToYyyyMmDd(row[8]), 
      endDate: formatDateToYyyyMmDd(row[9]), totalDays: row[10], status: row[18], pdfUrl: row[24]
    }));
    return {
      stats: { totalStaff: totalStaff > 0 ? totalStaff : 0, approved, pending, rejected },
      charts: {
        leaveTypeData: { labels: Object.keys(leaveTypeCounts), data: Object.values(leaveTypeCounts) },
        monthlyLeaveData: { labels: ['ม.ค.', 'ก.พ.', 'มี.ค.', 'เม.ย.', 'พ.ค.', 'มิ.ย.', 'ก.ค.', 'ส.ค.', 'ก.ย.', 'ต.ค.', 'พ.ย.', 'ธ.ค.'], data: monthlyLeaveCounts }
      },
      recentLeaves
    };
  } catch (e) { return null; }
}

function getLeaveHistory(filters) {
  try {
    const sheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
    if (sheet.getLastRow() <= 1) return [];
    const allData = sheet.getDataRange().getValues().slice(1);
    let filtered = filters.role === 'admin' ? allData : allData.filter(row => row[1] === filters.userId);

    if (filters.filterUserId && filters.filterUserId !== 'all') filtered = filtered.filter(row => row[1] === filters.filterUserId);
    if (filters.filterStartDate) {
        const sd = new Date(filters.filterStartDate); sd.setHours(0,0,0,0);
        filtered = filtered.filter(row => new Date(row[8]) >= sd);
    }
    if (filters.filterEndDate) {
        const ed = new Date(filters.filterEndDate);
        ed.setHours(23,59,59,999);
        filtered = filtered.filter(row => new Date(row[8]) <= ed);
    }
    
    return filtered.reverse().map(row => ({
        leaveId: row[0], requestDate: formatDateToYyyyMmDd(row[5]), leaveType: row[6],
        startDate: formatDateToYyyyMmDd(row[8]), endDate: formatDateToYyyyMmDd(row[9]),
        totalDays: row[10], status: row[18], pdfUrl: row[24], fullName: row[2] 
    }));
  } catch (e) { return null; }
}

function getLeaveSummaryReport(startDate, endDate) {
  try {
    const sheet = SPREADSHEET.getSheetByName(CONFIG.SHEET.LEAVE);
    if (sheet.getLastRow() <= 1) return [];
    const allData = sheet.getDataRange().getValues().slice(1);
    const summary = {};
    const fs = startDate ? new Date(startDate) : null; if(fs) fs.setHours(0,0,0,0);
    const fe = endDate ? new Date(endDate) : null; if(fe) fe.setHours(23,59,59,999);
    
    allData.forEach(row => {
      if (row[18] === 'อนุมัติ') {
        const d = new Date(row[8]);
        if ((!fs || d >= fs) && (!fe || d <= fe)) {
          const uid = row[1];
          if (!summary[uid]) summary[uid] = { fullName: row[2], position: row[3], totalLeaves: 0, totalDays: 0 };
          summary[uid].totalLeaves++;
          summary[uid].totalDays += (parseFloat(row[10]) || 0);
        }
      }
    });
    return Object.values(summary);
  } catch (e) { return null; }
}

function uploadBase64ImageToDrive(dataUrl, fileName) {
    const [meta, data] = dataUrl.split(',');
    const mimeType = meta.match(/:(.*?);/)[1];
    const blob = Utilities.newBlob(Utilities.base64Decode(data), mimeType, `${fileName}.png`);
    let folder, folders = DriveApp.getFoldersByName(CONFIG.SIGNATURE_FOLDER_NAME);
    if (folders.hasNext()) folder = folders.next();
    else folder = DriveApp.createFolder(CONFIG.SIGNATURE_FOLDER_NAME);
    const file = folder.createFile(blob);
    file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    return file.getId();
}

function formatDateToYyyyMmDd(dateValue) {
  if (!dateValue) return '';
  let d = (dateValue instanceof Date) ? dateValue : new Date(dateValue);
  if (isNaN(d.getTime())) return '';
  return Utilities.formatDate(d, "GMT+7", "yyyy-MM-dd");
}

function formatDateForPdf(dateValue) {
  if (!dateValue) return '......';
  try {
    const d = new Date(dateValue);
    return d.toLocaleDateString('th-TH', { day: 'numeric', month: 'long', year: 'numeric'});
  } catch (e) { return String(dateValue); }
}

function generatePdf(leaveInfo, approvalData, adminSignatureUrl, approvalDate) {
    const outputFolder = DriveApp.getFolderById(CONFIG.PDF_OUTPUT_FOLDER_ID);
    const templateFile = DriveApp.getFileById(CONFIG.SLIDE_TEMPLATE_ID);
    const newFileName = 'ใบลา_' + leaveInfo[2];
    const newPresentationFile = templateFile.makeCopy(newFileName, outputFolder);
    const presentation = SlidesApp.openById(newPresentationFile.getId());
    const slide = presentation.getSlides()[0];
    
    const finalComment = (approvalData.status === 'อนุมัติ') ? 'อนุมัติ' : `ไม่อนุมัติ เนื่องจาก ${approvalData.comment}`;
    const replacements = {
        '{{ชื่อโรงเรียน}}': leaveInfo[4], '{{วันที่ยื่นคำขอ}}': formatDateForPdf(leaveInfo[5]),
        '{{ประเภทการลา}}': leaveInfo[6], '{{ชื่อ-นามสกุล}}': leaveInfo[2], '{{ตำแหน่ง}}': leaveInfo[3], 
        '{{เหตุผลการลา}}': leaveInfo[7], '{{วันที่ลา}}': formatDateForPdf(leaveInfo[8]), 
        '{{วันสิ้นสุดการลา}}': formatDateForPdf(leaveInfo[9]), '{{จำนวนวันลา}}': leaveInfo[10], 
        '{{ชื่อครู}}': leaveInfo[25], '{{ชื่อวิชา}}': leaveInfo[26],
        '{{ประเภทการลารอบก่อน}}': leaveInfo[11], '{{วันที่ลารอบก่อน}}': formatDateForPdf(leaveInfo[12]), 
        '{{วันสิ้นสุดการลารอบก่อน}}': formatDateForPdf(leaveInfo[13]), '{{จำนวนการลารอบก่อน}}': leaveInfo[14], 
        '{{ที่ติดต่อระหว่างลา}}': leaveInfo[15], '{{หมายเลขโทรศัพท์}}': leaveInfo[16], 
        '{{ความเห็นผู้บังคับบัญชา}}': finalComment, '{{ชื่อ-นามสกุลผู้อนุมัติ}}': approvalData.adminUser.fullName, 
        '{{ตำแหน่งผู้อนุมัติ}}': approvalData.adminUser.position, '{{วันที่อนุมัติ}}': approvalDate
    };

    for (const k in replacements) slide.replaceAllText(k, replacements[k] || '......');
    insertSignature(slide, '{{ลายเซ็นครู}}', leaveInfo[17]);
    insertSignature(slide, '{{ลายเซ็นผู้อนุมัติ}}', adminSignatureUrl);

    presentation.saveAndClose();
    const pdfBlob = newPresentationFile.getAs(MimeType.PDF);
    const pdfFile = outputFolder.createFile(pdfBlob).setName(newFileName + '.pdf');
    pdfFile.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
    newPresentationFile.setTrashed(true);
    return pdfFile.getUrl();
}

function insertSignature(slide, placeholder, imageFileId) {
  try {
    const shape = findShapeByText(slide, placeholder);
    if (shape && imageFileId) {
      const imageFile = DriveApp.getFileById(imageFileId);
      shape.replaceWithImage(imageFile.getBlob());
    } else if (shape) shape.getText().setText('');
  } catch (e) { 
    const shape = findShapeByText(slide, placeholder);
    if(shape) try{ shape.getText().setText(''); }catch(err){} 
  }
}

function findShapeByText(slide, text) {
    const shapes = slide.getShapes();
    for (let i = 0; i < shapes.length; i++) {
        try { if (shapes[i].getText().asString().includes(text)) return shapes[i];
        } catch(e) {}
    }
    return null;
}
